% Explicacions:
% Primerament observem que amb trapezis hem aconseguit 3 xifres
% significatives, 3 xifres tamb significatives amb montecarlo i 11 xifres
% significatives amb el mtode de Romberg. Cal observar que per aconseguir
% les 3 xifres significatives que ha conseguit trapezis, montecarlo ha
% tingut que utilitzar una mostra molt gran, mirant la taula, alrededor de
% 10^8, montecarlo necessita una mostra molt gran per a conseguir prou
% xifres significatives. Y veiem com el mtode de Romberg es eficient,
% consegueix bastantes xifres significatives, fins i tot amb la frmula de
% Simpson, sent aquesta la columna dos. Millora el doble.
function r = romberg(f,a,b,n)
h = (b - a) ./ (2.^(0:n-1));
r(1,1) = (b - a) * (f(a) + f(b)) / 2;
for j = 2:n
subtotal = 0;
for i = 1:2^(j-2)
subtotal = subtotal + f(a + (2 * i - 1) * h(j));
end
r(j,1) = r(j-1,1) / 2 + h(j) * subtotal;
for k = 2:j
r(j,k) = (4^(k-1) * r(j,k-1) - r(j-1,k-1)) / (4^(k-1) - 1);
end
end
end
