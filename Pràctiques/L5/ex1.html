(1)
f = @(x)besselj(0,x);
x = 2; n = 5;
valor = -besselj(1,x);
hk = 1:n;
h(1:n) = 1./2.^hk;
for k = 1:n
ff(k) = (f(x+h(k))-f(x-h(k)))/(2*h(k));
end
er1 = abs(ff-valor);
V1 = [h(1:n); ff; er1]';
T = array2table(V1,'VariableNames',{'1/2^k','N_1 = Jo(2)','error'}); disp(T)
